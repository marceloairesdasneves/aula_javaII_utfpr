#if
if(condition) statement;
else statement;

#switch
switch(expression) {
    case constant-expression:
        statement(s);
        break; // optional
}

#for
for(initialization; condition; increment) {
    statement(s);
}

#while
while(condition) {
    statement(s);
}

#do-while
do {
    statement(s);
} while (condition);

#break
break;

#continue
continue;
